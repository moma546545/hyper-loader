    def _load_stats(self):
        try:
            if os.path.exists("xd_stats.json"):
                with open("xd_stats.json", "r") as f:
                    self.stats.update(json.load(f))
        except:
            pass

    def _save_stats(self):
        try:
            with open("xd_stats.json", "w") as f:
                json.dump(self.stats, f)
        except:
            pass

        if hasattr(self, 'stats_lbl'):

    def _draw_window_gradient(self, event=None):
        if not hasattr(self, "window_bg_canvas"):
            return
        try:
            if not self.winfo_exists() or not self.window_bg_canvas.winfo_exists():
                return
            c      = self.current_theme
            canvas = self.window_bg_canvas
            width  = max(1, self.winfo_width())
            height = max(1, self.winfo_height())
            canvas.delete("all")
            canvas.configure(bg=c["bg"])
            steps = 60
            for index in range(steps):
                factor = index / max(1, steps - 1)
                color  = self._blend(c["bg"], c["bg_2"], factor)
                y0 = int(height * index / steps)
                y1 = int(height * (index + 1) / steps)
                canvas.create_rectangle(0, y0, width, y1, outline="", fill=color)
            
            # Massive Aurora glowing orbs
            orb1  = self._blend(c["accent"], c["bg"], 0.85)
            orb2  = self._blend(c["gold"],   c["bg"], 0.88)
            orb3  = self._blend(c["accent_2"], c["bg"], 0.90)
            
            # Top right cyan/accent nebula
            canvas.create_oval(int(width*0.45), int(-height*0.25), int(width*1.2), int(height*0.55), outline="", fill=orb1, stipple="gray50")
            canvas.create_oval(int(width*0.55), int(-height*0.15), int(width*1.1), int(height*0.45), outline="", fill=self._blend(c["accent"], c["bg"], 0.75))
            
            # Bottom left pure glowing neon
            canvas.create_oval(int(-width*0.2), int(height*0.5), int(width*0.35), int(height*1.2), outline="", fill=orb2, stipple="gray50")
            canvas.create_oval(int(-width*0.1), int(height*0.6), int(width*0.25), int(height*1.1), outline="", fill=self._blend(c["gold"], c["bg"], 0.80))
            
            # Bottom right secondary accent mesh
            canvas.create_oval(int(width*0.75), int(height*0.7), int(width*1.15), int(height*1.1), outline="", fill=orb3)

            canvas.tk.call("lower", canvas._w)
        except tk.TclError:
            return

    def _draw_main_decor(self, event=None):
        if not hasattr(self, "main_decor_canvas"):
            return
        try:
            if not self.winfo_exists() or not self.main_container.winfo_exists() or not self.main_decor_canvas.winfo_exists():
                return
            c      = self.current_theme
            canvas = self.main_decor_canvas
            width  = max(1, self.main_container.winfo_width())
            height = max(1, self.main_container.winfo_height())
            canvas.delete("all")
            canvas.configure(bg=self._blend(c["bg_2"], c["panel"], 0.12))
            
            # Glassmorphism container background layer
            steps  = 50
            top    = self._blend(c["panel"], c["bg_2"], 0.10)
            bottom = self._blend(c["panel_alt"], c["bg"], 0.3)
            for index in range(steps):
                factor = index / max(1, steps - 1)
                color  = self._blend(top, bottom, factor)
                y0 = int(height * index / steps)
                y1 = int(height * (index + 1) / steps)
                canvas.create_rectangle(0, y0, width, y1, outline="", fill=color)
                
            # Inner container soft edge glows
            glow_cyan = self._blend(c["accent"], c["panel"], 0.92)
            glow_pink = self._blend(c["gold"], c["panel"], 0.94)

    def _animate_download_glow(self):
        c     = self.current_theme
        tick  = self._glow_tick % 40
        phase = tick / 39
        self._glow_tick += 1
        # Alternates between electric cyan and vibrant orange
        if phase < 0.5:
            glow = self._blend(c["accent"], c["gold"], phase * 0.6)
        else:
            glow = self._blend(c["gold"], c["accent"], (phase - 0.5) * 0.6)

        self.after(90, self._animate_download_glow)

    def _hex_to_rgb(self, value: str):
        value = value.lstrip("#")
        return tuple(int(value[i:i + 2], 16) for i in (0, 2, 4))

    def _rgb_to_hex(self, rgb):
        return "#" + "".join(f"{max(0, min(255, int(c))):02x}" for c in rgb)

    def _blend(self, start: str, end: str, factor: float):
        s = self._hex_to_rgb(start)
        e = self._hex_to_rgb(end)
        blended = (s[0] + (e[0] - s[0]) * factor, s[1] + (e[1] - s[1]) * factor, s[2] + (e[2] - s[2]) * factor)
        return self._rgb_to_hex(blended)

    def _apply_button_role(self, button, hovered: bool = False):
        role = self.button_roles.get(button, "ghost")
        c = self.current_theme
        try:
            state = str(button.cget("state"))
        except tk.TclError:
            return
        disabled = state == "disabled"
        if role == "primary":
            fg = self._blend(c["accent"], c["gold"], 0.08 if not hovered else 0.14)
            border = self._blend(c["accent_2"], c["gold"], 0.26 if not hovered else 0.38)
            text = "#ffffff" if not disabled else self._blend(c["text"], c["muted"], 0.45)
        elif role == "secondary":
            fg = self._blend(c["panel_soft"], c["accent"], 0.12 if not hovered else 0.18)
            border = self._blend(c["border"], c["accent"], 0.28 if not hovered else 0.4)
            text = c["text"] if not disabled else self._blend(c["text"], c["muted"], 0.45)
        else:
            fg = self._blend(c["panel"], c["bg"], 0.18 if not hovered else 0.1)
            border = self._blend(c["border"], c["gold"], 0.1 if not hovered else 0.22)
            text = c["text"] if not disabled else self._blend(c["text"], c["muted"], 0.45)
        if disabled:
            fg = self._blend(fg, c["panel"], 0.46)
            border = self._blend(border, c["panel"], 0.4)
        try:
            button.configure(
                fg_color=fg,
                hover_color=fg,
                text_color=text,
                border_color=border,
                border_width=2 if hovered and not disabled else 1,
            )
        except tk.TclError:
            return

    def _format_duration_label(self, total_seconds):
        if not isinstance(total_seconds, (int, float)) or total_seconds <= 0:
            return "--:--"
        total_seconds = int(total_seconds)
        hours = total_seconds // 3600
        minutes = (total_seconds % 3600) // 60
        seconds = total_seconds % 60
        if hours:
            return f"{hours:02d}:{minutes:02d}:{seconds:02d}"
        return f"{minutes:02d}:{seconds:02d}"

    def _format_bytes(self, value):
        if not isinstance(value, (int, float)) or value <= 0:
            return "--"
        units = ["B", "KB", "MB", "GB", "TB"]
        size = float(value)
        unit_index = 0
        while size >= 1024 and unit_index < len(units) - 1:
            size /= 1024
            unit_index += 1
        if unit_index == 0:
            return f"{int(size)} {units[unit_index]}"
        return f"{size:.1f} {units[unit_index]}"

    def apply_theme(self, theme_name: str, animated: bool = True):
        self.current_theme = self.themes.get(theme_name, self.themes[next(iter(self.themes))])
        if not animated:
            self.rendered_theme = self.current_theme.copy()
            self._apply_theme_colors(self.current_theme)
            return
        # تأثير انتقال الألوان الناعم
        steps = 10
        start_theme = self.rendered_theme.copy()
        end_theme = self.current_theme.copy()
        keys = ["bg", "bg_2", "panel", "panel_alt", "panel_soft", "accent", "accent_2", "gold", "text", "muted", "border"]

        for step in range(1, steps + 1):
            factor = step / steps
            mixed = {k: self._blend(start_theme[k], end_theme[k], factor) for k in keys}
            self.after(step * 15, lambda m=mixed: self._apply_theme_colors(m))
        self.rendered_theme = self.current_theme.copy()

    def append_log(self, text: str):
        level = "INFO"
        lower = text.lower()
        if any(keyword in lower for keyword in ["error", "failed", "خطأ"]):
            level = "ERROR"
        elif any(keyword in lower for keyword in ["100%", "completed", "done", "تم "]):
            level = "DONE"
        timestamp = datetime.now().strftime("%H:%M:%S")
        line = f"{timestamp}  {level:<5}  {text}\n"
        self.log_text.configure(state="normal")
        self.log_text.insert("end", line)
        self.log_text.see("end")
        self.log_text.configure(state="disabled")

    def _refresh_history_ui(self):
        if not hasattr(self, "history_scroll"):
            return
        for child in self.history_scroll.winfo_children():
            child.destroy()
        self.history_row_widgets = []
        term = self.history_search_var.get().strip().lower()
        rows = self.stats.get("download_history", [])
        for item in rows:
            text_blob = " ".join([
                str(item.get("url", "")),
                str(item.get("status", "")),
                str(item.get("mode", "")),
                str(item.get("format", "")),
                str(item.get("message", "")),
            ]).lower()
            if term and term not in text_blob:
                continue
            frame = ctk.CTkFrame(self.history_scroll, corner_radius=16, border_width=1)
            frame.pack(fill="x", padx=8, pady=6)
            frame.grid_columnconfigure(1, weight=1)
            status_text = item.get("status", "unknown")
            status = ctk.CTkLabel(frame, text=status_text, width=92, height=30, corner_radius=10, font=ctk.CTkFont(size=12, weight="bold"))
            status.grid(row=0, column=0, rowspan=2, padx=(10, 8), pady=10, sticky="w")
            title = ctk.CTkLabel(frame, text=f"{self._format_history_time(item.get('timestamp', ''))} • {item.get('mode', '--')} • {item.get('format', '--')}", anchor="e", justify="right", font=ctk.CTkFont(size=12, weight="bold"))
            title.grid(row=0, column=1, padx=8, pady=(10, 4), sticky="e")

    def _find_ffmpeg_location(self):
        ffmpeg_path = shutil.which("ffmpeg")
        if ffmpeg_path:
            return os.path.dirname(ffmpeg_path)
        candidates = glob.glob(os.path.join(os.path.expanduser("~"), "AppData", "Local", "Microsoft", "WinGet", "Packages", "yt-dlp.FFmpeg*", "**", "ffmpeg.exe"), recursive=True)
        if candidates:
            return os.path.dirname(candidates[0])
        return None

    def _toggle_playlist_item(self, index: int):
        if 0 <= index < len(self.playlist_check_vars):

    def _on_playlist_format_change(self, value):
        audio_formats = {"Original Audio", "MP3", "AAC", "M4A", "WAV", "AIFF", "WMA", "FLAC"}
        if value in audio_formats:
            self.mode_var.set("صوت")
            self.audio_format_var.set(value)
        else:
            self.mode_var.set("فيديو")
            self.video_format_var.set(value)
        self.refresh_mode_state()

    def _on_playlist_quality_change(self, value):
        self.video_quality_var.set(value)

    def _get_playlist_item_url(self, entry):
        video_id = entry.get("id")
        if video_id:
            return f"https://www.youtube.com/watch?v={video_id}"
        url = entry.get("url") or entry.get("webpage_url") or ""
        if url and not url.startswith("http") and entry.get("ie_key", "").lower() == "youtube":
            return f"https://www.youtube.com/watch?v={url}"
        return url

    def _get_quality_height(self, quality_text: str):
        mapping = {"8K": "4320", "4K": "2160"}
        cleaned = quality_text.split("(")[0].strip()
        if cleaned in mapping:
            return mapping[cleaned]
        return cleaned.replace("p", "").strip()

    def _format_total_duration(self, total_seconds: int):
        hours = total_seconds // 3600
        minutes = (total_seconds % 3600) // 60
        seconds = total_seconds % 60
        if hours:
            return f"{hours:02d}:{minutes:02d}:{seconds:02d}"
        return f"{minutes:02d}:{seconds:02d}"

    def _on_playlist_search_change(self, *args):
        self._apply_playlist_filter()

    def _apply_playlist_filter(self):
        query = self.playlist_search_var.get().strip().lower()
        for entry in self.playlist_entries:
            frame = entry.get("row")
            if frame is None:
                continue
            haystack = f"{entry.get('title', '')} {entry.get('url', '')}".lower()
            frame.pack_forget()
            if not query or query in haystack:
                frame.pack(fill="x", padx=15, pady=8)

    def _update_playlist_selection_ui(self):
        total = len(self.playlist_check_vars)
        selected_entries = [entry for entry, var in zip(self.playlist_entries, self.playlist_check_vars) if var.get()]
        selected_count = len(selected_entries)
        total_seconds = sum(int(entry.get("duration") or 0) for entry in selected_entries)
        self.playlist_selection_var.set(f"العناصر المحددة: {selected_count}/{total}")
        self.playlist_footer_var.set(f"المدة الإجمالية: {self._format_total_duration(total_seconds)} • الحجم التقديري: --")
        self.playlist_download_btn_var.set(f"تنزيل ({selected_count}/{total})")
        self.playlist_download_btn.configure(state="normal" if selected_count else "disabled")
        if total == 0:
            self.select_all_var.set(False)
        else:
            self.select_all_var.set(selected_count == total)

    def _preview_worker(self, url: str):
        cmd = [sys.executable, "-m", "yt_dlp", "--dump-single-json", "--skip-download", "--playlist-items", "1", url]
        try:
            res = subprocess.run(cmd, capture_output=True, text=True, encoding="utf-8", errors="replace")
            if res.returncode != 0:
                self.after(0, lambda: self.preview_meta_var.set("فشل التحقق من الرابط"))
                self.after(0, lambda: self.preview_title_var.set("خطأ في الاتصال"))
                self.after(0, lambda: self.preview_duration_var.set("المدة: --:--"))
                self.after(0, lambda: self.preview_source_var.set("المصدر: غير متاح"))
                self.after(0, lambda: self.preview_quality_var.set("التوصية: تحقق من الرابط"))
                self.after(0, lambda: self._set_preview_state("فشل التحليل", "error"))
                self.after(0, lambda: self._set_url_state("الرابط غير قابل للتحليل", "invalid"))
                return
            data = json.loads(res.stdout)
            if data.get("entries"):
                info = data["entries"][0]
            else:
                info = data
            title = info.get("title") or "عنوان مجهول"
            uploader = info.get("uploader") or info.get("channel") or "ناشر غير معروف"
            duration = info.get("duration") or 0
            meta = f"{uploader} • {self._format_duration_label(duration)}"
            thumb = info.get("thumbnail")
            source = info.get("extractor_key") or info.get("webpage_url_domain") or "ويب"
            resolution = info.get("resolution") or (f"{info.get('height')}p" if info.get("height") else "أفضل جودة")
            filesize = info.get("filesize_approx") or info.get("filesize")
            self.after(0, lambda: self.preview_title_var.set(title))
            self.after(0, lambda: self.preview_meta_var.set(meta))
            self.after(0, lambda: self.preview_duration_var.set(f"المدة: {self._format_duration_label(duration)}"))
            self.after(0, lambda: self.preview_source_var.set(f"المصدر: {source}"))
            quality_text = f"التوصية: {resolution}"
            if filesize:
                quality_text = f"{quality_text} • {self._format_bytes(filesize)}"
            self.after(0, lambda qt=quality_text: self.preview_quality_var.set(qt))
            self.after(0, lambda: self._set_preview_state("جاهز للتنزيل", "success"))
            self.after(0, lambda: self._set_url_state("الرابط جاهز للتنزيل", "valid"))
            if thumb:
                with urllib.request.urlopen(thumb, timeout=10) as response:
                    raw = response.read()
                image = Image.open(BytesIO(raw)).convert("RGB")
                image = image.resize((300, 168))
                ctk_img = ctk.CTkImage(light_image=image, dark_image=image, size=(300, 168))
                self.after(0, lambda img=ctk_img: self._apply_thumbnail(img))
        except Exception:
            self.after(0, lambda: self.preview_meta_var.set("تعذر تحميل بيانات المعاينة"))
            self.after(0, lambda: self.preview_title_var.set("فشل المعاينة"))
            self.after(0, lambda: self.preview_duration_var.set("المدة: --:--"))
            self.after(0, lambda: self.preview_source_var.set("المصدر: غير متاح"))
            self.after(0, lambda: self.preview_quality_var.set("التوصية: تحقق من الاتصال"))
            self.after(0, lambda: self._set_preview_state("فشل المعاينة", "error"))

        self.preview_thumb_label.configure(image=image, text="")
        self.preview_thumb_label.image = image

    def load_playlist(self):
        url = self.url_var.get().strip()
        if not url:
            messagebox.showerror("تنبيه", "الرابط مطلوب للبدء.")
            return
        if not self._check_yt_dlp():
            messagebox.showerror("تبعية مفقودة", "yt-dlp غير متوفر في النظام.")
            return
        self.select_frame("playlist")
        self.playlist_title_var.set("جاري تحميل القائمة...")
        self.playlist_footer_var.set("بعد اكتمال التحليل اختر المقاطع التي تريدها ثم اضغط تنزيل")
        self.playlist_download_btn_var.set("تنزيل (0/0)")
        self.playlist_download_btn.configure(state="disabled")
        self.playlist_loaded = False
        for frame in self.playlist_row_frames:
            frame.destroy()
        self.playlist_row_frames = []
        self.status_var.set("جلب قائمة التشغيل...")
        threading.Thread(target=self._load_playlist_worker, args=(url,), daemon=True).start()

    def _load_playlist_worker(self, url: str):
        cmd = [sys.executable, "-m", "yt_dlp", "--flat-playlist", "--dump-single-json", url]
        try:
            res = subprocess.run(cmd, capture_output=True, text=True, encoding="utf-8", errors="replace")
            if res.returncode != 0:
                self.after(0, lambda: self._load_playlist_failed(res.stderr.strip() or "فشل الجلب"))
                return
            data = json.loads(res.stdout)
            entries = data.get("entries") or []
            title = data.get("title") or "بدون اسم"
            self.after(0, lambda: self._render_playlist(title, entries))
        except Exception as exc:
            self.after(0, lambda: self._load_playlist_failed(str(exc)))

    def _load_playlist_failed(self, msg: str):
        self.status_var.set("فشل تحميل القائمة")
        self.playlist_footer_var.set("فشل تحميل القائمة. راجع الرابط ثم حاول مرة أخرى")
        self.playlist_download_btn.configure(state="disabled")
        self.append_log(f"خطأ: {msg}")
        self.playlist_loaded = False

    def _render_playlist(self, title: str, entries):
        for frame in self.playlist_row_frames:
            frame.destroy()
        self.playlist_output_format_var.set(self.video_format_var.get() if self.mode_var.get() == "فيديو" else self.audio_format_var.get())
        self.playlist_quality_var.set(self.video_quality_var.get())

        self.playlist_entries = []
        self.playlist_check_vars = []
        self.playlist_check_widgets = []
        self.playlist_row_frames = []
        self.playlist_progressbars = {}
        self.playlist_progresslabels = {}
        self.playlist_status_labels = {}
        self.playlist_title_labels = []
        self.playlist_url_labels = []
        self.playlist_meta_badges = []

        for idx, entry in enumerate(entries, start=1):
            video_title = entry.get("title") or f"مقطع رقم {idx}"
            video_id = entry.get("id")
            item_url = self._get_playlist_item_url(entry)
            self.playlist_entries.append({"index": idx, "title": video_title, "id": video_id, "url": item_url, "duration": entry.get("duration")})

            row = ctk.CTkFrame(self.playlist_scroll, corner_radius=18, border_width=1)
            row.pack(fill="x", padx=15, pady=8)
            row.grid_columnconfigure(1, weight=1)

            rhs = ctk.CTkFrame(row, fg_color="transparent")
            rhs.grid(row=0, column=2, sticky="e", padx=(10, 15), pady=14)
            var = tk.BooleanVar(value=False)
            cb = ctk.CTkCheckBox(rhs, text="", width=24, variable=var, corner_radius=6, command=self._on_playlist_item_toggle)
            cb.pack(side="right", padx=(10, 0))
            thumb_label = ctk.CTkLabel(rhs, text="...", width=80, height=46, corner_radius=12)
            thumb_label.pack(side="right")
            if video_id:
                threading.Thread(target=self._load_row_thumbnail, args=(video_id, thumb_label), daemon=True).start()

            mid = ctk.CTkFrame(row, fg_color="transparent")
            mid.grid(row=0, column=1, sticky="ew", padx=15, pady=14)
            title_lbl = ctk.CTkLabel(mid, text=video_title[:80], font=ctk.CTkFont(family=self.base_font_family, size=15, weight="bold"), anchor="e", justify="right")
            title_lbl.pack(fill="x")
            url_lbl = ctk.CTkLabel(mid, text=item_url[:80], font=ctk.CTkFont(family=self.base_font_family, size=11, weight="normal"), anchor="e", justify="right")
            url_lbl.pack(fill="x", pady=(2, 0))

            lhs = ctk.CTkFrame(row, fg_color="transparent")
            lhs.grid(row=0, column=0, sticky="w", padx=(15, 10), pady=14)
            index_lbl = ctk.CTkLabel(lhs, text=f"{idx:02d}", width=34, font=ctk.CTkFont(family=self.base_font_family, size=13, weight="bold"))
            index_lbl.pack(side="left", padx=(0, 10))
            status_lbl = ctk.CTkLabel(lhs, text="في الانتظار", width=98, height=28, corner_radius=11, font=ctk.CTkFont(family=self.base_font_family, size=12, weight="bold"))
            status_lbl.pack(side="left", padx=(0, 10))
            
            dur_sz_lbl = ctk.CTkLabel(lhs, text=self._format_duration_label(entry.get('duration', 0)), font=ctk.CTkFont(family=self.base_font_family, size=12, weight="bold"))
            dur_sz_lbl.pack(side="left")

            bar_frame = ctk.CTkFrame(row, fg_color="transparent")
            bar_frame.grid(row=1, column=0, columnspan=3, sticky="ew", padx=15, pady=(0, 14))
            bar_frame.grid_columnconfigure(0, weight=1)
            bar = ctk.CTkProgressBar(bar_frame, height=6, corner_radius=3)
            bar.set(0)
            bar.grid(row=0, column=0, sticky="ew", padx=(0, 10))
            lbl = ctk.CTkLabel(bar_frame, text="0%", font=ctk.CTkFont(family=self.base_font_family, size=11, weight="bold"))
            lbl.grid(row=0, column=1)

            row.bind("<Enter>", lambda _, r=row: self._on_row_enter(r))
            row.bind("<Leave>", lambda _, r=row: self._on_row_leave(r))
            for widget in [thumb_label, title_lbl, url_lbl, index_lbl, status_lbl, dur_sz_lbl, bar_frame]:
                widget.bind("<Button-1>", lambda _, item_index=idx - 1: self._toggle_playlist_item(item_index))
                widget.bind("<Enter>", lambda _, r=row: self._on_row_enter(r))
                widget.bind("<Leave>", lambda _, r=row: self._on_row_leave(r))

            self.playlist_check_vars.append(var)
            self.playlist_check_widgets.append(cb)
            self.playlist_row_frames.append(row)
            self.playlist_progressbars[idx] = bar
            self.playlist_progresslabels[idx] = lbl
            self.playlist_status_labels[idx] = status_lbl
            self.playlist_title_labels.append(title_lbl)
            self.playlist_url_labels.append(url_lbl)
            self.playlist_meta_badges.extend([dur_sz_lbl])

            self.status_var.set("وضع المقطع الواحد")
        self._update_playlist_selection_ui()
        self._apply_playlist_filter()
        self._refresh_playlist_widget_colors()

    def add_current_to_queue(self):
        url = self.url_var.get().strip()
        if not url:
            messagebox.showerror("تنبيه", "الرابط فارغ!")
            return
        if any(item["url"] == url for item in self.queue_items):
            messagebox.showinfo("تنبيه", "هذا الرابط مضاف بالفعل للطابور.")
            return
        title = self.preview_title_var.get() if self.preview_title_var.get() != "لا توجد معاينة حالياً" else url
        self.queue_items.append({"url": url, "title": title})
        self._refresh_queue()
        self.select_frame("queue")

    def clear_queue(self):
        self.queue_items = []
        self._refresh_queue()

    def move_queue_item(self, index: int, direction: int):
        target = index + direction
        if target < 0 or target >= len(self.queue_items):
            return
        self.queue_items[index], self.queue_items[target] = self.queue_items[target], self.queue_items[index]
        self._refresh_queue()

    def remove_queue_item(self, index: int):
        if 0 <= index < len(self.queue_items):
            self.queue_items.pop(index)
            self._refresh_queue()

            # Left actions

    def show_toast(self, title: str, message: str):
        try:
            toast = ctk.CTkToplevel(self)
            toast.overrideredirect(True)
            toast.attributes("-alpha", 0.0)
            toast.attributes("-topmost", True)
            c = self.current_theme
            toast.configure(fg_color=c["bg"])
            w, h = 340, 96
            sw = toast.winfo_screenwidth()
            sh = toast.winfo_screenheight()
            x = sw - w - 24
            y = sh - h - 64
            toast.geometry(f"{w}x{h}+{x}+{y+40}")
            frame = ctk.CTkFrame(toast, corner_radius=16, border_width=1, fg_color=c["panel_alt"], border_color=self._blend(c["accent"], c["gold"], 0.3))
            frame.pack(fill="both", expand=True, padx=4, pady=4)
            lbl_title = ctk.CTkLabel(frame, text=title, font=ctk.CTkFont(family=self.base_font_family, size=15, weight="bold"), text_color=c["text"], anchor="e", justify="right")
            lbl_title.pack(fill="x", padx=16, pady=(16, 2))
            lbl_msg = ctk.CTkLabel(frame, text=message, font=ctk.CTkFont(family=self.base_font_family, size=12), text_color=c["muted"], anchor="e", justify="right")
            lbl_msg.pack(fill="x", padx=16, pady=(0, 16))
            def animate_in(step=0):
                if step <= 15:
                    toast.attributes("-alpha", step/15.0)
                    toast.geometry(f"{w}x{h}+{x}+{y+40 - int(40*(step/15.0))}")
                    self.after(15, animate_in, step+1)
                else:
                    self.after(4000, fade_out)
            def fade_out(step=15):
                if step >= 0:
                    toast.attributes("-alpha", step/15.0)
                    self.after(15, fade_out, step-1)
                else:
                    toast.destroy()
            animate_in()
        except:
            pass

    def _notify_desktop(self, title: str, message: str):
        try:
            if self.desktop_notifier:
                self.desktop_notifier.show_toast(title, message, duration=5, threaded=True)
                return
        except Exception:
            pass
        self.show_toast(title, message)

    def _create_tray_image(self):
        icon_size = 64
        c = self.current_theme
        image = Image.new("RGBA", (icon_size, icon_size), (0, 0, 0, 0))
        draw = ImageDraw.Draw(image)
        draw.rounded_rectangle((4, 4, 60, 60), radius=14, fill=c["panel_alt"], outline=c["accent"], width=4)
        draw.text((18, 18), "XD", fill=c["gold"])
        return image

    def _setup_system_tray(self):
        if not pystray:
            return
        try:
            menu = pystray.Menu(
                pystray.MenuItem("إظهار", self._tray_show_window),
                pystray.MenuItem("خروج", self._tray_quit_app),
            )
            self.tray_icon = pystray.Icon("x_downloader_premium", self._create_tray_image(), "X-Downloader Premium", menu)
            self.tray_thread = threading.Thread(target=self.tray_icon.run, daemon=True)
            self.tray_thread.start()
            self.tray_ready = True
        except Exception:
            self.tray_icon = None
            self.tray_ready = False

    def _tray_show_window(self, icon=None, item=None):
        self.after(0, self._restore_from_tray)

    def _tray_quit_app(self, icon=None, item=None):
        self.after(0, self._quit_app)

    def _restore_from_tray(self):
        try:
            self.deiconify()
            self.lift()
            self.focus_force()
            self.status_var.set("تمت استعادة النافذة")
        except Exception:
            pass

    def _handle_close_request(self):
        if self.tray_ready and self.tray_icon and not self.is_quitting:
            self.withdraw()
            self._notify_desktop("X-Downloader Premium", "التطبيق يعمل الآن في الخلفية بجوار الساعة.")
            return
        self._quit_app()

    def _quit_app(self):
        self.is_quitting = True
        try:
            if self.tray_icon:
                self.tray_icon.stop()
        except Exception:
            pass
        self.destroy()

    def start_queue(self):
        if self.is_downloading:
            messagebox.showinfo("النظام مشغول", "هناك عملية تحميل نشطة حالياً.")
            return
        if not self.queue_items:
            self.start_download()
            return
        self.queue_running = True
        self._start_next_queue_item()
    def _start_next_queue_item(self):
        if not self.queue_running:
            return
        if self.is_downloading:
            return
        if not self.queue_items:
            self.queue_running = False
            self.status_var.set("انتهى الطابور")
            self.append_log("تم إنهاء جميع العمليات في الطابور.")
            post_action = self.post_action_var.get()
            if post_action == "إغلاق الجهاز":
                os.system("shutdown /s /t 60")
                self.append_log("سيتم إغلاق الجهاز بعد 60 ثانية...")
            elif post_action == "وضع السكون":
                os.system("rundll32.exe powrprof.dll,SetSuspendState 0,1,0")
                self.append_log("تم تفعيل وضع السكون.")
            return
        next_item = self.queue_items.pop(0)
        self._refresh_queue()
        self.url_var.set(next_item["url"])
        self.start_download(from_queue=True)

    def cancel_download(self):
        if not self.is_downloading:
            return
        self.cancel_requested = True
        self.status_var.set("جاري الإلغاء...")
        self.progress_info_var.set("إلغاء التحميل...")
        self.append_log("تم طلب إلغاء عملية التحميل الحالية.")
        process = self.current_download_process
        if process and process.poll() is None:
            try:
                process.terminate()
            except Exception:
                pass
        self._notify_desktop("X-Downloader Premium", "تم إرسال طلب الإلغاء.")

    def start_download(self, from_queue: bool = False):
        if self.is_downloading:
            messagebox.showinfo("النظام مشغول", "يرجى الانتظار حتى انتهاء التحميل الحالي.")
            return
        url = self.url_var.get().strip()
        out_dir = self.output_dir_var.get().strip()
        if not url:
            messagebox.showerror("خطأ", "الرابط مطلوب.")
            return
        if not out_dir:
            messagebox.showerror("خطأ", "لم يتم تحديد مجلد الحفظ.")
            return
        if not os.path.isdir(out_dir):
            os.makedirs(out_dir, exist_ok=True)
        if not self._check_yt_dlp():
            messagebox.showerror("خطأ فني", "yt-dlp غير متوفر في المسار.")
            return

        selected_indices = []
        output_format = self.playlist_output_format_var.get() if self.playlist_loaded else (self.video_format_var.get() if self.mode_var.get() == "فيديو" else self.audio_format_var.get())
        quality_text = self.playlist_quality_var.get() if self.playlist_loaded else self.video_quality_var.get()
        subtitle_value = self.subtitle_var.get()
        audio_formats = {"Original Audio", "MP3", "AAC", "M4A", "WAV", "AIFF", "WMA", "FLAC"}
        mode = "صوت" if output_format in audio_formats else "فيديو"

        if self.playlist_loaded and self.playlist_check_vars:
            for idx, var in enumerate(self.playlist_check_vars, start=1):
                if var.get():
                    selected_indices.append(idx)
            if not selected_indices:
                messagebox.showerror("خطأ", "يرجى اختيار مقطع واحد على الأقل.")
                return

        self.global_progress.set(0)
        self.progress_info_var.set("تحضير المحرك...")
        self.is_downloading = True
        self.cancel_requested = False
        self.current_download_process = None
        self.status_var.set("جاري التحميل...")
        self.active_download_context = {
            "timestamp": datetime.now().isoformat(),
            "url": url,
            "mode": mode,
            "format": output_format,
            "quality": quality_text,
            "subtitle": subtitle_value,
            "from_queue": from_queue,
        }
        self._notify_desktop("X-Downloader Premium", "بدأ التحميل الآن.")
        self._set_loading_state(True)
        threading.Thread(target=self._download_worker, args=(url, out_dir, selected_indices, from_queue, mode, output_format, quality_text, subtitle_value, dict(self.active_download_context)), daemon=True).start()

    def _download_worker(self, url: str, out_dir: str, selected_indices, from_queue: bool, mode: str, output_format: str, quality_text: str, subtitle_value: str, job_context: dict):
        output_template = os.path.join(out_dir, "%(title)s [%(id)s].%(ext)s")
        cmd = [sys.executable, "-m", "yt_dlp", "-o", output_template]
        cmd.extend(["--newline", "--no-warnings"])
        ffmpeg_location = self._find_ffmpeg_location()
        if ffmpeg_location:
            cmd.extend(["--ffmpeg-location", ffmpeg_location])
        if selected_indices:
            cmd.extend(["--playlist-items", ",".join(str(i) for i in selected_indices)])
            
        aria2c_path = shutil.which("aria2c")
        if aria2c_path:
            cmd.extend(["--downloader", "aria2c", "--downloader-args", "aria2c:-x 16 -s 16 -j 16 -k 1M"])


        subtitle_map = {
            "Arabic": "ar",
            "English": "en",
            "French": "fr",
            "German": "de",
            "Italian": "it",
            "Spanish": "es",
            "Swedish": "sv",
        }

        if mode == "فيديو":
            quality = self._get_quality_height(quality_text)
            fmt = output_format.lower()
            cmd.extend(["-f", f"bestvideo[height<={quality}]+bestaudio/best[height<={quality}]", "--merge-output-format", fmt])
            subtitle_code = subtitle_map.get(subtitle_value)
            if subtitle_code:
                cmd.extend(["--write-sub", "--write-auto-sub", "--sub-langs", f"{subtitle_code}.*", "--embed-subs", "--compat-options", "all"])
        else:
            if output_format == "Original Audio":
                cmd.extend(["-f", "bestaudio"])
            else:
                aq = self.audio_quality_var.get().replace("kbps", "K")
                af = output_format.lower()
                cmd.extend(["-f", "bestaudio", "-x", "--audio-format", af, "--audio-quality", aq])
            if output_format in ["MP3", "M4A", "FLAC"]:
                cmd.extend(["--embed-thumbnail", "--embed-metadata", "--parse-metadata", "title:%(artist)s - %(title)s", "--postprocessor-args", "ffmpeg:-loglevel error -hide_banner"])
        start_t = self.start_time_var.get().strip()
        end_t = self.end_time_var.get().strip()
        if start_t or end_t:
            cmd.extend(["--download-sections", f"*{start_t or '00:00'}-{end_t or 'inf'}"])

        cmd.extend(["--concurrent-fragments", "3"])
        cmd.append(url)

        current_item = selected_indices[0] if selected_indices else 1
        max_attempts = 3
        success = False
        cancelled = False
        last_error = ""
        attempts_used = 0
        try:
            for attempt in range(1, max_attempts + 1):
                attempts_used = attempt
                if self.cancel_requested:
                    cancelled = True
                    break
                if attempt > 1:
                    self.after(0, lambda a=attempt: self.append_log(f"محاولة إعادة {a}/{max_attempts}"))
                process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, universal_newlines=True, encoding="utf-8", errors="replace")
                self.current_download_process = process
                for line in process.stdout:
                    clean = line.strip()
                    if not clean:
                        continue
                    self.after(0, lambda t=clean: self.append_log(t))
                    if "error" in clean.lower() or "failed" in clean.lower() or "خطأ" in clean.lower():
                        last_error = clean
                    m_item = re.search(r"Downloading item (\d+) of (\d+)", clean)
                    if m_item and selected_indices:
                        pos = int(m_item.group(1)) - 1
                        if 0 <= pos < len(selected_indices):
                            current_item = selected_indices[pos]
                    m_pct = re.search(r"\[download\]\s+(\d+(?:\.\d+)?)%", clean)
                    if m_pct:
                        pct = float(m_pct.group(1))
                        m_speed = re.search(r"at\s+([~0-9\.]+[kMG]iB/s)", clean)
                        m_eta = re.search(r"ETA\s+([\d:]+)", clean)
                        speed_str = m_speed.group(1) if m_speed else "--"
                        eta_str = m_eta.group(1) if m_eta else "--:--"
                        self.after(0, lambda idx=current_item, p=pct, s=speed_str, e=eta_str: self._progress_for_item(idx, p, s, e))
                    if "100%" in clean and "[download]" in clean:
                        self.after(0, lambda idx=current_item: self._mark_item_done(idx))
                ret = process.wait()
                self.current_download_process = None
                if self.cancel_requested:
                    cancelled = True
                    break
                if ret == 0:
                    success = True
                    break
                if attempt < max_attempts:
                    wait_seconds = attempt * 2
                    self.after(0, lambda s=wait_seconds: self.append_log(f"فشل المحاولة، إعادة خلال {s} ثانية..."))
                    for _ in range(wait_seconds * 10):
                        if self.cancel_requested:
                            cancelled = True
                            break
                        time.sleep(0.1)
                    if cancelled:
                        break
            result_meta = dict(job_context)
            result_meta["attempts"] = attempts_used
            result_meta["cancelled"] = cancelled
            result_meta["error"] = last_error
            self.after(0, lambda s=success, fq=from_queue, m=result_meta: self._download_done(s, fq, m))
        except Exception as exc:
            result_meta = dict(job_context)
            result_meta["attempts"] = attempts_used or 1
            result_meta["cancelled"] = False
            result_meta["error"] = str(exc)
            self.after(0, lambda: self.append_log(f"خطأ فادح: {exc}"))
            self.after(0, lambda fq=from_queue, m=result_meta: self._download_done(False, fq, m))

    def _download_done(self, success: bool, from_queue: bool, meta: dict | None = None):
        meta = meta or {}
        self.is_downloading = False
        self.current_download_process = None
        self._set_loading_state(False)
        cancelled = bool(meta.get("cancelled"))
        self.global_progress.set(1 if success else 0)
        if cancelled:
            self.queue_running = False
            self.status_var.set("تم إلغاء التحميل")
            self.append_log("تم إلغاء التحميل بواسطة المستخدم.")
            self.show_toast("إلغاء", "تم إيقاف التحميل الحالي.")
            self._notify_desktop("X-Downloader Premium", "تم إلغاء التحميل.")
            history_entry = {
                "timestamp": meta.get("timestamp", datetime.now().isoformat()),
                "url": meta.get("url", self.url_var.get().strip()),
                "mode": meta.get("mode", self.mode_var.get()),
                "format": meta.get("format", "--"),
                "status": "cancelled",
                "message": "تم إيقاف العملية يدويًا",
                "attempts": meta.get("attempts", 1),
            }
            self._record_download_history(history_entry)
            return
        if success:
            self.status_var.set("اكتملت العملية")
            self.append_log("تم تنزيل الملفات بنجاح.")

            if meta.get("mode", self.mode_var.get()) == "فيديو":
                self.stats["total_videos"] += 1
            else:
                self.stats["total_audios"] += 1
            self._update_stats_ui()
            history_entry = {
                "timestamp": meta.get("timestamp", datetime.now().isoformat()),
                "url": meta.get("url", self.url_var.get().strip()),
                "mode": meta.get("mode", self.mode_var.get()),
                "format": meta.get("format", "--"),
                "status": "success",
                "message": f"تم التنزيل بنجاح بعد {meta.get('attempts', 1)} محاولة",
                "attempts": meta.get("attempts", 1),
            }
            self._record_download_history(history_entry)

            if not from_queue:
                self.show_toast("نجاح", "اكتمل تنزيل جميع الملفات المختارة.")
                self._notify_desktop("X-Downloader Premium", "اكتمل التحميل بنجاح.")
        else:
            self.status_var.set("فشل التحميل")
            self.append_log("تعذر إتمام عملية التنزيل.")
            history_entry = {
                "timestamp": meta.get("timestamp", datetime.now().isoformat()),
                "url": meta.get("url", self.url_var.get().strip()),
                "mode": meta.get("mode", self.mode_var.get()),
                "format": meta.get("format", "--"),
                "status": "failed",
                "message": meta.get("error") or "فشل بعد إعادة المحاولة",
                "attempts": meta.get("attempts", 1),
            }
            self._record_download_history(history_entry)
            if not from_queue:
                self.show_toast("خطأ", "حدثت مشكلة أثناء التحميل. راجع السجل.")
                self._notify_desktop("X-Downloader Premium", "فشل التحميل. راجع السجل.")
        if self.queue_running:
            self.after(500, self._start_next_queue_item)
        elif from_queue and success:
            self.show_toast("اكتمال الطابور", "تم تنزيل كافة الملفات من الطابور بنجاح.")
            self._notify_desktop("X-Downloader Premium", "تم إنهاء الطابور بالكامل.")
