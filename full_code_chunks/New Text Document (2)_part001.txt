    def _prune_inactive_workers(self):
        if not hasattr(self, "active_workers"):
            return 0
        stale_keys = []
        for wid, worker in list(self.active_workers.items()):
            if worker is None:
                stale_keys.append(wid)
                continue
            try:
                if worker.isFinished():
                    stale_keys.append(wid)
            except RuntimeError:
                stale_keys.append(wid)
        for wid in stale_keys:
            worker = self.active_workers.pop(wid, None)
            if self.current_worker is worker:
                self.current_worker = None
        if self._display_progress_wid not in self.active_workers:
            self._display_progress_wid = next(iter(self.active_workers.keys()), None)
        return len(stale_keys)

    def _active_workers_count(self):
        self._prune_inactive_workers()
        return sum(1 for w in self.active_workers.values() if w.isRunning())

    def _on_formats_probe_finished(self, success: bool, output: str, error: str):
        worker = self._formats_worker
        self._formats_worker = None
        if worker is not None:
            try:
                worker.deleteLater()
            except RuntimeError:
                pass
        if not success:
            self._warn(error or _("تعذر جلب الصيغ من yt-dlp"))
            return
        text = str(output or "").strip()
        if not text:
            self._warn(_("لم يتم استرجاع أي صيغ من yt-dlp"))
            return
        framed = f"{'=' * 28}\nyt-dlp -F\n{'=' * 28}\n{text}\n{'=' * 28}"
        self.search_view.log_text.append(framed)
        self._info(_("تم عرض الصيغ المتاحة في سجل التحليل"))

    def _check_app_updates(self, background: bool):
        if self._app_update_check_worker is not None and self._app_update_check_worker.isRunning():
            return
        manifest_url = str(os.getenv("VIDDOWNLOADER_UPDATE_MANIFEST_URL", "") or "").strip()
        if not manifest_url:
            if not background:
                self._warn("Update manifest URL غير مضبوط. عيّن VIDDOWNLOADER_UPDATE_MANIFEST_URL")
            return
        if not background:
            self._append_log("[Update] Checking for app updates...")

        def _on_done(ok: bool, manifest: dict, error: str):
            self._app_update_check_worker = None
            if not ok:
                if not background:
                    self._warn(f"تعذر فحص التحديث: {error}")
                return
            if not isinstance(manifest, dict) or not manifest.get("available"):
                if not background:
                    self._info("لا يوجد تحديث متاح حالياً")
                return
            self._available_app_update = manifest
            latest = str(manifest.get("version", "") or "").strip()
            notes = str(manifest.get("notes", "") or "").strip()
            msg = f"يتوفر تحديث جديد: {latest}\nإصدارك الحالي: {APP_VERSION}"
            if notes:
                msg = msg + "\n\n" + notes[:1200]
            resp = QMessageBox.question(
                self,
                "تحديث البرنامج",
                msg + "\n\nهل تريد تنزيل التحديث الآن؟",
                QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No,
                QMessageBox.StandardButton.Yes,
            )
            if resp == QMessageBox.StandardButton.Yes:
                self._download_app_update(manifest)

        worker = AppUpdateCheckWorker(APP_VERSION, manifest_url)
        worker.finished.connect(_on_done)
        self._app_update_check_worker = worker
        worker.start()

    def _download_app_update(self, manifest: dict):
        if self._app_update_download_worker is not None and self._app_update_download_worker.isRunning():
            self._warn("تنزيل التحديث جارٍ بالفعل")
            return
        if not isinstance(manifest, dict):
            self._warn("بيانات التحديث غير صالحة")
            return
        platform_block = manifest.get("windows") if sys.platform == "win32" else manifest.get("linux")
        if not isinstance(platform_block, dict):
            platform_block = manifest
        url = str(platform_block.get("url", "") or "").strip()
        sha256 = str(platform_block.get("sha256", "") or "").strip()
        version = str(manifest.get("version", "") or "update").strip()
        if not url:
            self._warn("Manifest لا يحتوي على رابط تنزيل صالح")
            return
        if not re.fullmatch(r"[A-Fa-f0-9]{64}", sha256 or ""):
            self._warn("Manifest لا يحتوي على SHA256 صالح للتحديث")
            return
        updates_dir = os.path.join(os.path.dirname(self._session_path()), "updates")
        out_path = os.path.join(updates_dir, f"VidDownloader-{version}.zip")
        self._available_app_update_download_path = out_path
        self._append_log(f"[Update] Downloading update: {version}")

        def _on_progress(downloaded: int, total: int):
            try:
                if total > 0:
                    pct = int(round((downloaded / total) * 100))
                    self.search_view.progress_bar.setValue(max(0, min(100, pct)))
            except (TypeError, ValueError, ZeroDivisionError, OverflowError):
                pass

        def _on_done(ok: bool, path: str, error: str):
            self._app_update_download_worker = None
            if not ok:
                self._warn(f"تعذر تنزيل التحديث: {error}")
                return
            self._append_log("[Update] Update downloaded successfully.")
            resp = QMessageBox.question(
                self,
                "تحديث جاهز",
                "تم تنزيل التحديث بنجاح.\nهل تريد تثبيته الآن وإعادة تشغيل البرنامج؟",
                QMessageBox.StandardButton.Yes | QMessageBox.StandardButton.No,
                QMessageBox.StandardButton.Yes,
            )
            if resp == QMessageBox.StandardButton.Yes:
                self._install_downloaded_app_update(path)

        worker = AppUpdateDownloadWorker(url, sha256, out_path)
        worker.progress.connect(_on_progress)
        worker.finished.connect(_on_done)
        self._app_update_download_worker = worker
        worker.start()

    def _install_downloaded_app_update(self, zip_path: str):
        zip_path = str(zip_path or "").strip()
        if not zip_path or not os.path.isfile(zip_path):
            self._warn("ملف التحديث غير موجود")
            return
        if not bool(getattr(sys, "frozen", False)):
            self._warn("التحديث التلقائي يعمل فقط على النسخة المعبأة (Packaging)")
            return
        exe_path = str(sys.executable)
        install_dir = os.path.dirname(exe_path)
        if not install_dir:
            self._warn("تعذر تحديد مسار التثبيت")
            return
        script_path = os.path.join(tempfile.gettempdir(), f"viddl_apply_update_{uuid.uuid4().hex}.ps1")
        script = r"""
param(
  [Parameter(Mandatory=$true)][string]$ZipPath,
  [Parameter(Mandatory=$true)][string]$InstallDir,
  [Parameter(Mandatory=$true)][string]$ExeName,
  [Parameter(Mandatory=$true)][int]$ParentPid
)
try { Wait-Process -Id $ParentPid -ErrorAction SilentlyContinue } catch {}
$tmp = Join-Path $env:TEMP ("viddl-upd-" + [guid]::NewGuid().ToString("N"))
New-Item -ItemType Directory -Path $tmp -Force | Out-Null
Expand-Archive -LiteralPath $ZipPath -DestinationPath $tmp -Force
$root = $tmp
$items = Get-ChildItem -LiteralPath $tmp
if ($items.Count -eq 1 -and $items[0].PSIsContainer) { $root = $items[0].FullName }
$rcArgs = @($root, $InstallDir, "/E", "/NFL", "/NDL", "/NJH", "/NJS", "/NC", "/NS", "/NP", "/R:2", "/W:1")
& robocopy @rcArgs | Out-Null
$destExe = Join-Path $InstallDir $ExeName
if (Test-Path -LiteralPath $destExe) {
  Start-Process -FilePath $destExe -WorkingDirectory $InstallDir
}
Remove-Item -LiteralPath $ZipPath -Force -ErrorAction SilentlyContinue
Remove-Item -LiteralPath $tmp -Recurse -Force -ErrorAction SilentlyContinue
"""
        try:
            with open(script_path, "w", encoding="utf-8") as f:
                f.write(script.strip() + "\n")
        except OSError as exc:
            self._warn(f"تعذر كتابة سكربت التحديث: {exc}")
            return
        try:
            pid = int(os.getpid())
            exe_name = os.path.basename(exe_path)
            subprocess.Popen(
                [
                    "powershell.exe",
                    "-NoProfile",
                    "-ExecutionPolicy",
                    "Bypass",
                    "-File",
                    script_path,
                    "-ZipPath",
                    zip_path,
                    "-InstallDir",
                    install_dir,
                    "-ExeName",
                    exe_name,
                    "-ParentPid",
                    str(pid),
                ],
                creationflags=getattr(subprocess, "CREATE_NO_WINDOW", 0),
            )
        except (OSError, subprocess.SubprocessError, ValueError) as exc:
            self._warn(f"تعذر تشغيل سكربت التحديث: {exc}")
            return
        QApplication.instance().quit()

    def _play_previous_in_history(self):
        items = self.stats.get("download_history", [])
        if not items or not hasattr(self, "_current_playing_url"):
            return
        current_idx = -1
        for i, item in enumerate(items):
            if item.get("url") == self._current_playing_url:
                current_idx = i
                break
        if current_idx > 0:
            self._play_history_item(items[current_idx - 1])

    def _play_next_in_history(self):
        items = self.stats.get("download_history", [])
        if not items or not hasattr(self, "_current_playing_url"):
            return
        current_idx = -1
        for i, item in enumerate(items):
            if item.get("url") == self._current_playing_url:
                current_idx = i
                break
        if current_idx != -1 and current_idx < len(items) - 1:
            self._play_history_item(items[current_idx + 1])

    def _find_best_file_path(self, item: dict) -> str:
        if not isinstance(item, dict):
            return ""
        raw_path = str(item.get("file_path", "") or "").strip()
        if raw_path and os.path.isfile(raw_path):
            return raw_path
        search_dirs = []
        for candidate in [
            self.current_download_path,
            (self.search_view.get_out_dir() if hasattr(self, "search_view") and hasattr(self.search_view, "get_out_dir") else (self.search_view.out_dir_input.text().strip() if hasattr(self, "out_dir_input") else "")),
            default_download_dir(),
        ]:
            c = str(candidate or "").strip()
            if c and os.path.isdir(c) and c not in search_dirs:
                search_dirs.append(c)
        if not search_dirs:
            return ""
        title = str(item.get("title", "") or "").strip().lower()
        url = str(item.get("url", "") or "").strip().lower()
        ids = []
        for pattern in [
            r"\[([A-Za-z0-9_-]{6,})\]",
            r"[?&]v=([A-Za-z0-9_-]{6,})",
            r"youtu\.be/([A-Za-z0-9_-]{6,})",
        ]:
            for m in re.findall(pattern, f"{title} {url}"):
                if m not in ids:
                    ids.append(m)
        media_exts = {
            ".mp4", ".webm", ".mkv", ".avi", ".mov", ".flv",
            ".mp3", ".m4a", ".aac", ".wav", ".flac", ".ogg", ".wma",
        }
        max_candidates_per_dir = 2000
        title_tokens = [t for t in re.findall(r"[a-zA-Z0-9\u0600-\u06FF]+", title) if len(t) > 2][:6]
        for folder in search_dirs:
            try:
                files = []
                with os.scandir(folder) as entries:
                    for entry in entries:
                        if len(files) >= max_candidates_per_dir:
                            break
                        if not entry.is_file():
                            continue
                        ext = os.path.splitext(entry.name)[1].lower()
                        if ext in media_exts:
                            files.append(entry.path)
            except OSError:
                continue
            if ids:
                for fp in files:
                    name = os.path.basename(fp).lower()
                    if any(f"[{vid.lower()}]" in name or vid.lower() in name for vid in ids):
                        return fp
            if title_tokens:
                ranked = []
                for fp in files:
                    name = os.path.basename(fp).lower()
                    score = sum(1 for t in title_tokens if t in name)
                    if score > 0:
                        ranked.append((score, fp))
                if ranked:
                    ranked.sort(key=lambda x: (x[0], os.path.getmtime(x[1])), reverse=True)
                    return ranked[0][1]
        return ""

    def _start_conversion(self):
        if self._conversion_worker is not None and self._conversion_worker.isRunning():
            QMessageBox.information(self, "Busy", "A conversion is already in progress.")
            return
        input_path = self.tools_view.conv_input.text()
        if not input_path or not os.path.exists(input_path):
            QMessageBox.warning(self, "Error", "Please select a valid file.")
            return
        
        fmt = self.tools_view.conv_fmt.currentText().lower()
        output_path = os.path.splitext(input_path)[0] + f"_converted.{fmt}"
        
        ffmpeg_bin = shutil.which("ffmpeg") or "ffmpeg"
        cmd = [ffmpeg_bin, "-y", "-i", input_path, output_path]
        worker = ConversionWorker(cmd, output_path, self)
        self._conversion_worker = worker
        self._append_log(f"Conversion started: {input_path} -> {output_path}")
        self._set_status("جاري التحويل")

        def _on_conversion_done(success: bool, final_path: str, error: str):
            self._conversion_worker = None
            if success:
                QMessageBox.information(self, "Done", f"Conversion finished successfully.\n{final_path}")
                self._append_log(f"Conversion finished: {final_path}")
                self._set_status("اكتمل التحويل")
                return
            msg = str(error or "FFmpeg conversion failed.").strip()
            QMessageBox.critical(self, "Error", msg)
            self._append_log(f"[Conversion ERROR] {msg}")
            self._set_status("فشل التحويل")

        worker.finished.connect(_on_conversion_done)
        worker.start()

    def _normalize_search_history(self, values):
        out = []
        seen = set()
        now = datetime.now().timestamp()
        ttl_seconds = max(1, int(self.search_history_ttl_days)) * 24 * 3600
        
        for raw in (values or []):
            if isinstance(raw, dict):
                url = str(raw.get("url", "")).strip()
                ts = float(raw.get("timestamp", now))
            else:
                url = str(raw or "").strip()
                ts = now
            
            if not url or not url.startswith("http"):
                continue
                
            if now - ts > ttl_seconds:
                continue
                
            low = url.lower()
            if low in seen:
                continue
            seen.add(low)
            out.append({"url": url, "timestamp": ts})
        return out[: self._search_history_limit]

    def _record_search_history(self, value: str):
        text = str(value or "").strip()
        if not text.startswith("http"):
            return
        entry = {"url": text, "timestamp": datetime.now().timestamp()}
        merged = [entry] + self.search_history
        self.search_history = self._normalize_search_history(merged)
        self.search_history_model.setStringList([e["url"] for e in self.search_history])
        if hasattr(self, "_search_history_save_timer") and self._search_history_save_timer is not None:
            self._search_history_save_timer.start()
        else:
            self._save_session()

    def _estimate_size_label(self, duration_seconds: int, quality_label: str):
        q = str(quality_label).lower()
        if "8k" in q:
            per_minute_mb = 450
        elif "2160" in q or "4k" in q:
            per_minute_mb = 220
        elif "1440" in q:
            per_minute_mb = 140
        elif "1080" in q:
            per_minute_mb = 90
        elif "720" in q:
            per_minute_mb = 50
        elif "480" in q:
            per_minute_mb = 28
        elif "360" in q:
            per_minute_mb = 17
        elif "240" in q:
            per_minute_mb = 10
        else:
            per_minute_mb = 7
        mb = (int(duration_seconds or 0) / 60) * per_minute_mb
        if mb >= 1024:
            return f"{mb/1024:.1f} GB"
        return f"{mb:.1f} MB"

    def _storage_guard_paths(self) -> list[str]:
        seen = set()
        paths = []
        for wid in list(self.active_workers.keys()):
            task = self.queue_manager.get_task(wid) if isinstance(wid, int) else None
            out_dir = str((task or {}).get("out_dir", "")).strip()
            if out_dir and out_dir not in seen:
                seen.add(out_dir)
                paths.append(out_dir)
        fallback = str(self.current_download_path or "").strip()
        if not fallback and hasattr(self, "out_dir_input"):
            fallback = self.search_view.out_dir_input.text().strip()
        if fallback and fallback not in seen:
            paths.append(fallback)
        return paths

    def _pause_downloads_for_storage_guard(self, message: str):
        self.queue_paused = True
        try:
            self.queue_manager.pause_queue()
        except (RuntimeError, AttributeError) as exc:
            logger.debug(f"تعذر إيقاف الطابور بسبب Storage Guard: {exc}")
        for wid, worker in list(self.active_workers.items()):
            self.pause_requested_workers.add(wid)
            task = self.queue_manager.get_task(wid) if isinstance(wid, int) else None
            if isinstance(task, dict):
                task["status"] = "paused"
                task["next_retry_at"] = 0
            try:
                worker.stop()
            except (RuntimeError, AttributeError) as exc:
                logger.debug(f"تعذر إيقاف worker {wid} بسبب Storage Guard: {exc}")
        if message != self._storage_guard_last_message or not self._storage_guard_alerted:
            self._append_log(message)
            self._warn(message)
            self._show_tray_message(
                "SnapDownloader",
                message[:120],
                QSystemTrayIcon.MessageIcon.Warning,
                3500,
            )
        self._storage_guard_alerted = True
        self._storage_guard_last_message = message
        self._set_status(_("المساحة منخفضة"))
        self._save_session()
        self._refresh_downloads_list()

    def _on_queue_reorder_requested(self, from_row: int, to_row: int):
        if self.downloads_filter not in {"active", "queued", "scheduled"}:
            return
        src = self.downloads_view.downloads_model.get_item(int(from_row))
        if not isinstance(src, dict):
            return
        src_index = src.get("queue_index")
        if not isinstance(src_index, int):
            return
        if to_row is None:
            return
        to_row = int(to_row)
        if to_row < 0:
            to_index = self.queue_manager.get_item_count() - 1
        else:
            dst = self.downloads_view.downloads_model.get_item(to_row)
            dst_index = dst.get("queue_index") if isinstance(dst, dict) else None
            if not isinstance(dst_index, int):
                return
            to_index = dst_index
        if to_index < 0:
            return
        moved = self.queue_manager.move_task(src_index, to_index)
        if not moved:
            return
        self._save_session()
        self._refresh_downloads_list()

    def _allow_private_bulk_urls(self) -> bool:
        return str(os.getenv("SNAPDOWNLOADER_ALLOW_PRIVATE_BULK_URLS", "")).strip().lower() in {"1", "true", "yes", "on"}

    def _is_allowed_bulk_url(self, url: str) -> bool:
        try:
            parsed = urllib.parse.urlparse(str(url or "").strip())
        except ValueError:
            return False
        if parsed.scheme not in {"http", "https"}:
            return False
        hostname = str(parsed.hostname or "").strip().lower()
        if not hostname:
            return False
        if self._allow_private_bulk_urls():
            return True
        if hostname == "localhost" or hostname.endswith(".local"):
            return False
        try:
            ip = ipaddress.ip_address(hostname)
        except ValueError:
            return True
        return not (
            ip.is_private
            or ip.is_loopback
            or ip.is_link_local
            or ip.is_reserved
            or ip.is_multicast
            or ip.is_unspecified
        )

    def _extract_urls(self, text: str):
        urls = []
        rejected_count = 0
        for raw in str(text or "").replace("\r", "\n").split("\n"):
            line = raw.strip()
            if not line:
                continue
            if "://" not in line and line.startswith("www."):
                line = "https://" + line
            if self._is_allowed_bulk_url(line):
                urls.append(line)
            else:
                rejected_count += 1
        seen = set()
        unique = []
        for url in urls:
            if url in seen:
                continue
            seen.add(url)
            unique.append(url)
        return unique, rejected_count

    def _bulk_settings_fallback(self) -> dict:
        mode = "video" if hasattr(self.browser_view, "bulk_video_radio") and self.browser_view.bulk_video_radio.isChecked() else "audio"
        fmt = self.browser_view.bulk_format_combo.currentText() if hasattr(self.browser_view, "bulk_format_combo") else "MP4"
        quality = self.browser_view.bulk_quality_combo.currentText() if hasattr(self.browser_view, "bulk_quality_combo") else "8K"
        subtitle = self.browser_view.bulk_subtitle_combo.currentText() if hasattr(self.browser_view, "bulk_subtitle_combo") else "None"
        return {
            "mode": str(mode or "audio"),
            "format": str(fmt or "MP4"),
            "quality": str(quality or "8K"),
            "subtitle": str(subtitle or "None"),
        }

    def _analyze_bulk(self, links=None, settings=None):
        provided_links = links if isinstance(links, list) else None
        if provided_links is None:
            text = self.browser_view.get_bulk_text() if hasattr(self, "browser_view") and hasattr(self.browser_view, "get_bulk_text") else self.browser_view.bulk_text.toPlainText()
            urls, rejected_count = self._extract_urls(text)
        else:
            normalized_text = "\n".join(str(line or "").strip() for line in provided_links if str(line or "").strip())
            urls, rejected_count = self._extract_urls(normalized_text)
        if rejected_count > 0:
            self._warn(_("تم تجاهل {count} رابط غير صالح أو محلي").format(count=rejected_count))
        if not urls:
            self._warn(_("لا توجد روابط"))
            return
        if self._metadata_fetch_worker is not None:
            self._warn(_("يوجد تحليل روابط جارٍ بالفعل"))
            return
        fallback_settings = self._bulk_settings_fallback()
        if isinstance(settings, dict):
            raw_bulk_settings = dict(settings)
        elif hasattr(self, "browser_view") and hasattr(self.browser_view, "get_bulk_settings"):
            raw_bulk_settings = self.browser_view.get_bulk_settings()
        else:
            raw_bulk_settings = {}
        bulk_settings = {
            "mode": str(raw_bulk_settings.get("mode") or fallback_settings["mode"]),
            "format": str(raw_bulk_settings.get("format") or fallback_settings["format"]),
            "quality": str(raw_bulk_settings.get("quality") or fallback_settings["quality"]).replace(" kbps", "kbps"),
            "subtitle": str(raw_bulk_settings.get("subtitle") or fallback_settings["subtitle"]),
        }
        self._bulk_analysis_queue = list(urls)
        self._bulk_analysis_results = []
        self._bulk_analysis_settings = {
            "format": bulk_settings["format"],
            "quality": bulk_settings["quality"],
            "subtitle": bulk_settings["subtitle"],
            "mode": bulk_settings["mode"],
            "total_urls": len(urls),
        }
        if hasattr(self, "browser_view") and hasattr(self.browser_view, "set_bulk_analyze_state"):
            self.browser_view.set_bulk_analyze_state(False, _("جاري التحليل {done}/{total}").format(done=0, total=len(urls)))
        else:
            self.bulk_analyze_btn.setEnabled(False)
            self.bulk_analyze_btn.setText(_("جاري التحليل {done}/{total}").format(done=0, total=len(urls)))
        self._append_log(_("بدء تحليل {count} رابط قبل إضافتهم للطابور").format(count=len(urls)))
        self._process_bulk_analysis_queue()

    def _build_bulk_task_from_metadata(self, url: str, title: str, thumbnail: str, duration_seconds: int, channel: str = "") -> dict:
        settings = dict(self._bulk_analysis_settings or {})
        task = self._build_task(
            url=url,
            title=title or url,
            thumbnail=thumbnail or "",
            fmt=settings.get("format", "MP4"),
            quality=settings.get("quality", "8K"),
        )
        task["mode"] = settings.get("mode", "video")
        task["format"] = settings.get("format", task.get("format", "MP4"))
        task["quality"] = settings.get("quality", task.get("quality", "8K"))
        task["subtitle"] = settings.get("subtitle", task.get("subtitle", "None"))
        task["duration_seconds"] = int(duration_seconds or 0)
        task["channel"] = str(channel or "").strip()
        task["status"] = "pending"
        return task

    def _process_bulk_analysis_queue(self):
        if self._metadata_fetch_worker is not None:
            return
        if not self._bulk_analysis_queue:
            self._finalize_bulk_analysis()
            return
        processed = len(self._bulk_analysis_results)
        total_urls = int(self._bulk_analysis_settings.get("total_urls", 0) or 0)
        progress_text = _("جاري التحليل {done}/{total}").format(done=min(processed, total_urls), total=total_urls)
        if hasattr(self, "browser_view") and hasattr(self.browser_view, "set_bulk_analyze_state"):
            self.browser_view.set_bulk_analyze_state(False, progress_text)
        else:
            self.bulk_analyze_btn.setText(progress_text)
        url = self._bulk_analysis_queue.pop(0)
        self._metadata_fetch_worker = AnalyzeWorker(url)
        self._metadata_fetch_worker.finished.connect(
            lambda success, message, payload, items, target_url=url: self._on_bulk_analysis_finished(
                target_url,
                success,
                message,
                payload,
                items,
            )
        )
        self._metadata_fetch_worker.start()

    def _on_bulk_analysis_finished(self, source_url: str, success: bool, message: str, payload: dict, items: list):
        self._metadata_fetch_worker = None
        generated_tasks = []
        if success and isinstance(payload, dict) and payload.get("kind") == "playlist" and items:
            for entry in items:
                generated_tasks.append(
                    self._build_bulk_task_from_metadata(
                        url=str(entry.get("url", "")).strip() or source_url,
                        title=str(entry.get("title", "")).strip() or source_url,
                        thumbnail=str(entry.get("thumbnail", "")).strip(),
                        duration_seconds=int(entry.get("duration_seconds") or 0),
                    )
                )
            self._append_log(_("تم تحليل قائمة تشغيل وإضافة {count} عنصر").format(count=len(generated_tasks)))
        elif success and isinstance(payload, dict):
            generated_tasks.append(
                self._build_bulk_task_from_metadata(
                    url=str(payload.get("webpage_url", "")).strip() or source_url,
                    title=str(payload.get("title", "")).strip() or source_url,
                    thumbnail=str(payload.get("thumbnail", "")).strip(),
                    duration_seconds=int(payload.get("duration_seconds") or 0),
                    channel=str(payload.get("channel", "")).strip(),
                )
            )
            self._append_log(
                _("تم تحليل الرابط: {title}").format(title=generated_tasks[0].get("title", source_url))
            )
        else:
            generated_tasks.append(
                self._build_bulk_task_from_metadata(
                    url=source_url,
                    title=source_url,
                    thumbnail="",
                    duration_seconds=0,
                )
            )
            self._append_log(
                _("تعذر جلب بيانات كاملة للرابط، تمت إضافته كرابط مباشر: {url}").format(url=source_url)
            )
        self._bulk_analysis_results.extend(generated_tasks)
        total_urls = int(self._bulk_analysis_settings.get("total_urls", 0) or 0)
        processed_urls = total_urls - len(self._bulk_analysis_queue)
        progress_text = _("جاري التحليل {done}/{total}").format(done=processed_urls, total=total_urls)
        if hasattr(self, "browser_view") and hasattr(self.browser_view, "set_bulk_analyze_state"):
            self.browser_view.set_bulk_analyze_state(False, progress_text)
        else:
            self.bulk_analyze_btn.setText(progress_text)
        QTimer.singleShot(0, self._process_bulk_analysis_queue)

    def _finalize_bulk_analysis(self):
        tasks = list(self._bulk_analysis_results)
        total_urls = int(self._bulk_analysis_settings.get("total_urls", 0) or 0)
        if hasattr(self, "browser_view") and hasattr(self.browser_view, "set_bulk_analyze_state"):
            self.browser_view.set_bulk_analyze_state(True, _("تحليل وإضافة للطابور"))
        else:
            self.bulk_analyze_btn.setEnabled(True)
            self.bulk_analyze_btn.setText(_("تحليل وإضافة للطابور"))
        self._metadata_fetch_worker = None
        self._bulk_analysis_queue = []
        self._bulk_analysis_results = []
        self._bulk_analysis_settings = {}
        if not tasks:
            self._warn(_("لم يتم تجهيز أي عناصر صالحة للطابور"))
            return
        self.queue_manager.add_tasks(tasks)
        self._save_session()
        if hasattr(self, "browser_view") and hasattr(self.browser_view, "clear_bulk_text"):
            self.browser_view.clear_bulk_text()
        else:
            self.browser_view.bulk_text.clear()
        self._append_log(
            _("تم تحليل {urls} رابط وإضافة {tasks} عنصر إلى الطابور").format(
                urls=total_urls, tasks=len(tasks)
            )
        )
        self._switch_view("downloads")
        self._set_downloads_filter("queued")
